CREATE TABLE Commercials_BK (
 CM_Id varchar(50),
 BankName varchar(100) PRIMARY KEY NOT NULL,
 Type varchar(50),
 CharterType varchar(50),
 Headquarters varchar(100),
 FailureDate varchar (50),
 InsuranceFund varchar(50),
 TransactionType varchar(50),
 TotalDeposits varchar(50),
 TotalAssets varchar(50)
);
use Banks;
CREATE TABLE Failed_BK (
 FB_Id VARCHAR(50) PRIMARY KEY NOT NULL,
 CERT VARCHAR (50) FOREIGN KEY REFERENCES InstitutesAll(CERT),
 ReportingDate VARCHAR(50),
 BankName VARCHAR(100) FOREIGN KEY REFERENCES Commercials_BK (BankName),
 CITY VARCHAR(50),
 COUNTY VARCHAR(50),
 STATE VARCHAR (50),
 TotalOffices VARCHAR(50),
 TotalEmployees VARCHAR(50),
 TotalAssets VARCHAR(50),
 NetLoansLeases VARCHAR(50),
 NoncurrentLloansLeases VARCHAR(50),
 BankPremisesFixedAssets VARCHAR (50),
 GoodwillOtherIntangibles VARCHAR (50),
 TotalLiabilities VARCHAR(50),
 TotalDeposits VARCHAR(50),
 InterestBearingDeposits VARCHAR (50),
 BankEquityCapital VARCHAR (50),
 UndividedProfits VARCHAR (50),
 NumberCommercialBanks VARCHAR(50),
 NetInterestMargin VARCHAR (50),
 PrimeLoanRate VARCHAR(50),
 FederalFundsRate VARCHAR(50),
 GDPPerCapita VARCHAR(50),
 GDPGrowthPerCapitaQuarterToQuarter VARCHAR(50),
 GDPInBillions VARCHAR(50),
 GDPGrowthQuarterToQuarter VARCHAR(50),
 ClosingDate VARCHAR(50),
 Failure VARCHAR(50)
);
use Banks;
CREATE TABLE InstitutesAll(
   INST_Id VARCHAR(50),
   CERT VARCHAR(50) PRIMARY KEY NOT NULL,
   BankName VARCHAR(100) FOREIGN KEY REFERENCES Commercials_BK (BankName),
   ADDRESS VARCHAR(100),
   CITY VARCHAR(50),
   COUNTY VARCHAR(50),
   STATE VARCHAR(50),
   ZIP VARCHAR(50),
   TotalOffices VARCHAR(50),
   Type VARCHAR(50),
   StartDate VARCHAR(50),
   TotalAssets VARCHAR(50),
   DepartmentID VARCHAR(50),
   TotalEquity VARCHAR(50),
   ROA VARCHAR(50),
   ROAQ VARCHAR(50),
   ROE VARCHAR(50),
   ROEQ VARCHAR(50),
   ROAPTX VARCHAR(50),
   ROAPTXQ VARCHAR(50),
   BankNetInc VARCHAR(50),
   InsuranceFund VARCHAR (50),
   BankWebSite VARCHAR(50)
   );
